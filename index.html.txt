        function filterAll() {
            let searchText = document.getElementById('search-box').value.toUpperCase();
            let selectedCategory = document.getElementById('category-select').value;
            let items = document.getElementsByClassName('proverb-item');

            for (let item of items) {
                // –°–º–µ–Ω–∞ —è–∑—ã–∫–∞
                let textElement = item.querySelector('.proverb-text');
                if (currentLang === 'all') {
                    textElement.innerText = item.getAttribute('data-text-ru');
                } else {
                    let trans = item.getAttribute('data-text-' + currentLang);
                    textElement.innerText = trans ? trans : item.getAttribute('data-text-ru');
                }

                // –§–∏–ª—å—Ç—Ä—ã
                let content = item.innerText.toUpperCase(); // –ò—â–µ–º –≤–æ –≤—Å–µ–º —Ç–µ–∫—Å—Ç–µ –∫–∞—Ä—Ç–æ—á–∫–∏
                let matchSearch = content.includes(searchText);
               
                let itemLangs = item.getAttribute('data-langs');
                let matchLang = (currentLang === 'all') || (itemLangs && itemLangs.includes(currentLang));
               
                let itemCategory = item.getAttribute('data-category');
                let matchCategory = (selectedCategory === 'all') || (itemCategory === selectedCategory);

                if (matchSearch && matchLang && matchCategory) {
                    item.style.display = "";
                } else {
                    item.style.display = "none";
                }
            }
        }

        // –ó–ê–ü–£–°–ö –ü–†–ò –°–¢–ê–†–¢–ï

        renderProverbs();
